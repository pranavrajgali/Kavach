### 🔁 Client-side Polling Pattern
Use a polling loop in Streamlit to fetch task states:
```python
import streamlit as st
import requests
import time

job_id = st.text_input("Enter Job ID")
if job_id:
    status_placeholder = st.empty()
    while True:
        res = requests.get(f"http://localhost:8000/status/{job_id}").json()
        status = res.get("status")
        status_placeholder.write(f"Current Status: **{status}**")
        if status in ["COMPLETED", "FAILED"]:
            break
        time.sleep(2)
```

### 🔴 SHAP Text Token Highlighter
Render Smali tokens in HTML based on attribution values:
```python
def render_shap_tokens(tokens, scores):
    # tokens: List[str], scores: List[float] (-1.0 to 1.0)
    html_str = "<div style='line-height:2;'>"
    for token, score in zip(tokens, scores):
        if score > 0.1: # Malicious
            bg_color = f"rgba(239, 68, 68, {min(score, 0.8)})" # Transparent Red
            html_str += f"<span style='background-color:{bg_color}; padding:2px 4px; border-radius:4px; margin:2px;' title='SHAP: {score:.3f}'>{token}</span>"
        elif score < -0.1: # Benign
            bg_color = f"rgba(59, 130, 246, {min(abs(score), 0.8)})" # Transparent Blue
            html_str += f"<span style='background-color:{bg_color}; padding:2px 4px; border-radius:4px; margin:2px;' title='SHAP: {score:.3f}'>{token}</span>"
        else:
            html_str += f"<span>{token}</span> "
    html_str += "</div>"
    st.markdown(html_str, unsafe_allow_html=True)
```

### 🛡️ Frida Subprocess Spawn Code

```python
import subprocess

def run_frida_bypass(package_name, script_path):
    # Command: frida -U -f com.example.app -l frida_bypass.js --no-pause
    cmd = [
        "frida", "-U", "-f", package_name,
        "-l", script_path, "--no-pause"
    ]
    process = subprocess.Popen(cmd, stdout=subprocess.PIPE, stderr=subprocess.PIPE)
    return process
```
